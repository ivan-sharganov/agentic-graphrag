###############
# OWL Restrictions (requirements)
###############

# Each Encounter must be linked to exactly one Traveller
pcfr:Encounter rdfs:subClassOf [
  a owl:Restriction ;
  owl:onProperty pcfr:hasTraveler ;
  owl:qualifiedCardinality "1"^^xsd:nonNegativeInteger ;
  owl:onClass pcfr:Traveller
] .

# Each DocumentCheck must have exactly one requestedDocument (DocumentInstance)
pcfr:DocumentCheck rdfs:subClassOf [
  a owl:Restriction ;
  owl:onProperty pcfr:requestedDocument ;
  owl:qualifiedCardinality "1"^^xsd:nonNegativeInteger ;
  owl:onClass pcfr:DocumentInstance
] .

# Each Questioning must have exactly one topicKey (string)
pcfr:Questioning rdfs:subClassOf [
  a owl:Restriction ;
  owl:onProperty pcfr:topicKey ;
  owl:qualifiedCardinality "1"^^xsd:nonNegativeInteger ;
  owl:onDataRange xsd:string
] .
